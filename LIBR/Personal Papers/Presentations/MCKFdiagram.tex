% Column 2 - Estimation of the State

% Column 3 - State covariance computation
\node[draw, align=center] (StateCov) [red, right of=stateEST, xshift=3cm] {\textbf{State covariance at $k$} \\ $P_{k-1|k-1}$};
\node[draw, align=center] (statePREDcov) [gold, below of=StateCov] {\textbf{State prediction covariance} \\ $P_{k|k-1}\approx$ \\ $\frac{1}{S-1} \sum_{i=1}^{S} (f(x_k)-\hat{x}_{k-1|k-1                      })\mathbf{1}^T$}; 	
\node[draw, align=center] (InnoCOV) [gold, below of=statePREDcov] {\textbf{Innovation Covariance} \\ $P^{zz}_{k|k-1}\approx$ \\ $\frac{1}{S-1} \sum_{i=1}^{S} (h(x_k)-\hat{z}_{k|k-1                      })\mathbf{1}^T$ };
\node[draw, align=center] (CrossCOV) [gold, below of=InnoCOV] {\textbf{Cross Covariance} \\ $P^{xz}_{k|k-1}\approx$ \\  $\frac{1}{S-1} \sum_{i=1}^{S} (f(x_k)-\hat{x}_{k-1|k-1                      })(h(x_k)-\hat{z}_{k|k-1                      })$ };
\node[draw, align=center] (FilterG) [goldenRod, below of=CrossCOV,xshift=-2.5cm] {\textbf{Filter Gain} \\ $W_k=$ \\  $P^{xx}_{k|k-1} (P^{zz}_{k|k-1})^{-1}$ };
\node[draw, align=center] (updateCOV) [red, below of=CrossCOV, yshift=-2cm] {\textbf{Updated State Covariance} \\ $P_{k|k} = $ \\ $P_{k|k}-W_k P^{zz}_{k|k-1} W_k^{T}$ };
